FROM(
	SELECT
		r.i,
		r.rword
	FROM (
	SELECT
		row_number() OVER (
			ORDER BY cnt DESC
		) AS i,

		LOWER(rword) AS rword,

		cnt
	FROM
		new_srch_frmla_20181108_simil_word
	WHERE

		removed = false
		) as r
	where
		r.i > 125000
